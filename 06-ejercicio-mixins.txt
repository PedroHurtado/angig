class Mixin{
    constructor(Base){
        this.Base = Base
    }
    withMixins(...mixins){
        return mixins.reduceRight((p,m)=>m(p),this.Base)

        //Base+add
        //Base+add+update
        //Base+add+update+remove

    }
}
